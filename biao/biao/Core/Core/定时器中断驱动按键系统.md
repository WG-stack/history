# 定时器中断驱动按键系统设计

## 1. 设计优势分析

### 1.1 当前按键方案问题
- **轮询方式**: 主循环中轮询按键状态，占用CPU资源
- **响应延迟**: 依赖主循环周期，按键响应不够及时
- **电源管理**: 无法有效降低功耗
- **实时性差**: 无法保证按键检测的实时性

### 1.2 定时器中断方案优势
- **定时扫描**: 固定周期扫描，保证响应一致性
- **低功耗**: 可以在扫描间隔进入低功耗模式
- **实时性好**: 定时器中断优先级可配置
- **资源节省**: 减少主循环负担

### 3.2 按键扫描实现
```c
#include "key_scan_algorithm.h"
#include "isr_event_handler.h"

// 全局扫描上下文
static KeyScanContext_TypeDef g_key_scan_context = {0};

// 读取按键矩阵状态
static uint16_t ReadKeyMatrix(void) {
    // 设置行输出，读取列输入
    GPIO_InitTypeDef GPIO_InitStruct = {0};
    
    // 配置行引脚为输出
    GPIO_InitStruct.Pin = KEY_ROW_PINS;
    GPIO_InitStruct.Mode = GPIO_MODE_OUTPUT_PP;
    GPIO_InitStruct.Pull = GPIO_NOPULL;
    GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_HIGH;
    HAL_GPIO_Init(KEY_GPIO_PORT, &GPIO_InitStruct);
    
    uint16_t matrix_state = 0;
    
    // 扫描每一行
    for (uint8_t row = 0; row < KEY_ROW_COUNT; row++) {
        // 激活当前行
        HAL_GPIO_WritePin(KEY_GPIO_PORT, KEY_ROW_PINS, ~(1 << row));
        
        // 短暂延时
        HAL_DelayMicroseconds(1);
        
        // 读取列状态
        uint16_t col_state = HAL_GPIO_ReadPin(KEY_GPIO_PORT, KEY_COL_PINS);
        
        // 将列状态添加到矩阵状态
        matrix_state |= (col_state << (row * KEY_COL_COUNT));
        
        // 恢复行引脚为高电平
        HAL_GPIO_WritePin(KEY_GPIO_PORT, KEY_ROW_PINS, GPIO_PIN_SET);
    }
    
    return matrix_state;
}

// 按键状态机处理
static KeyScanResult_TypeDef ProcessKeyState(uint16_t current_state) {
    KeyScanResult_TypeDef result = {0};
    
    switch (g_key_scan_context.state) {
        case KEY_SCAN_STATE_IDLE:
            if (current_state != g_key_scan_context.stable_state) {
                g_key_scan_context.debounce_count = 0;
                g_key_scan_context.state = KEY_SCAN_STATE_DEBOUNCING;
                g_key_scan_context.key_press_start_time = HAL_GetTick();
            }
            break;
            
        case KEY_SCAN_STATE_DEBOUNCING:
            if (current_state == g_key_scan_context.stable_state) {
                // 防抖完成
                g_key_scan_context.state = KEY_SCAN_STATE_IDLE;
            } else if (current_state != g_key_scan_context.last_state) {
                // 按键状态变化，重置防抖计数
                g_key_scan_context.debounce_count = 0;
                g_key_scan_context.key_press_start_time = HAL_GetTick();
            } else {
                g_key_scan_context.debounce_count++;
                if (g_key_scan_context.debounce_count >= KEY_DEBOUNCE_TIME_MS / KEY_SCAN_INTERVAL_MS) {
                    // 防抖完成，检测按键类型
                    if (current_state != 0) {
                        // 有按键按下
                        result.key_type = DetectKeyFromMatrix(current_state);
                        result.event = KEY_EVENT_PRESS;
                        g_key_scan_context.state = KEY_SCAN_STATE_PRESSED;
                        g_key_scan_context.long_press_count = 0;
                    }
                    g_key_scan_context.stable_state = current_state;
                }
            }
            break;
            
        case KEY_SCAN_STATE_PRESSED:
            if (current_state == 0) {
                // 按键释放
                result.key_type = DetectKeyFromMatrix(g_key_scan_context.stable_state);
                result.event = KEY_EVENT_RELEASE;
                g_key_scan_context.state = KEY_SCAN_STATE_RELEASE_DEBOUNCING;
                g_key_scan_context.key_release_start_time = HAL_GetTick();
            } else {
                // 持续按压，检测长按
                g_key_scan_context.long_press_count++;
                if (g_key_scan_context.long_press_count >= KEY_LONG_PRESS_TIME_MS / KEY_SCAN_INTERVAL_MS) {
                    result.key_type = DetectKeyFromMatrix(current_state);
                    result.event = KEY_EVENT_LONG_PRESS;
                    g_key_scan_context.is_long_press_detected = true;
                }
            }
            break;
            
        case KEY_SCAN_STATE_LONG_PRESS:
            // 长按状态下的重复扫描
            if (current_state == 0) {
                g_key_scan_context.state = KEY_SCAN_STATE_RELEASE_DEBOUNCING;
                g_key_scan_context.key_release_start_time = HAL_GetTick();
            }
            break;
            
        case KEY_SCAN_STATE_RELEASE_DEBOUNCING:
            if (current_state == 0) {
                g_key_scan_context.state = KEY_SCAN_STATE_IDLE;
                g_key_scan_context.stable_state = 0;
            } else {
                g_key_scan_context.debounce_count++;
                if (g_key_scan_context.debounce_count >= KEY_DEBOUNCE_TIME_MS / KEY_SCAN_INTERVAL_MS) {
                    g_key_scan_context.state = KEY_SCAN_STATE_PRESSED;
                    g_key_scan_context.stable_state = current_state;
                }
            }
            break;
    }
    
    g_key_scan_context.last_state = current_state;
    return result;
}

// 按键类型检测
static Key_TypeDef DetectKeyFromMatrix(uint16_t matrix_state) {
    // 根据矩阵状态确定按下了哪个按键
    // 这里需要根据具体的硬件连接来实现
    
    switch (matrix_state) {
        case KEY_MATRIX_ON_ZERO:    return KEY_ON_ZERO;
        case KEY_MATRIX_DOWN_ABS:   return KEY_DOWN_ABS_SET;
        case KEY_MATRIX_UP_IN_MM:   return KEY_UP_IN_MM;
        case KEY_MATRIX_TOL:        return KEY_TOL;
        case KEY_MATRIX_M:          return KEY_M;
        case KEY_MATRIX_ABS_ZERO:   return KEY_ABS_ZERO_SWITCH;
        default:
            return KEY_NONE;
    }
}
```

## 4. 定时器中断服务程序

### 4.1 中断服务程序
```c
// 定时器中断服务程序
void KEY_SCAN_TIMER_IRQHandler(void) {
    // 清除定时器中断标志
    HAL_TIM_IRQHandler(&hkey_scan_timer);
    
    // 执行按键扫描
    KeyTimer_ScanISR();
}

// 按键扫描ISR
void KeyTimer_ScanISR(void) {
    // 1. 快速读取按键矩阵状态
    uint16_t current_state = ReadKeyMatrix();
    
    // 2. 处理按键状态机
    KeyScanResult_TypeDef scan_result = ProcessKeyState(current_state);
    
    // 3. 检查是否有有效事件
    if (scan_result.event != KEY_EVENT_NONE) {
        // 4. 创建事件数据
        Event_TypeDef key_event = {
            .type = EVENT_KEY_PRESS,
            .priority = EVENT_PRIORITY_HIGH,
            .data.key_event.key_result = scan_result,
            .data.key_event.press_duration = 
                (g_key_scan_context.state == KEY_SCAN_STATE_PRESSED) ?
                (HAL_GetTick() - g_key_scan_context.key_press_start_time) : 0,
            .data.key_event.is_long_press = g_key_scan_context.is_long_press_detected,
            .timestamp = HAL_GetTick(),
            .source_id = SOURCE_ID_KEY_TIMER
        };
        
        // 5. 中断安全地发布事件
        ISR_PostEvent(EVENT_KEY_PRESS, &key_event.data);
        
        // 6. 触发事件处理
        ISR_EventQueue_TriggerProcessing();
        
        // 7. 重置长按检测标志
        g_key_scan_context.is_long_press_detected = false;
    }
}
```

### 5.1 扫描算法优化
```c
// 优化的按键矩阵扫描
static inline uint16_t ReadKeyMatrix_Optimized(void) {
    // 使用位操作优化
    uint16_t matrix_state = 0;
    
    // 批量配置行引脚
    GPIO_InitTypeDef GPIO_InitStruct = {0};
    GPIO_InitStruct.Pin = KEY_ROW_PINS;
    GPIO_InitStruct.Mode = GPIO_MODE_OUTPUT_PP;
    GPIO_InitStruct.Pull = GPIO_NOPULL;
    GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_HIGH;
    
    // 一次性配置所有行
    HAL_GPIO_Init(KEY_GPIO_PORT, &GPIO_InitStruct);
    
    for (uint8_t row = 0; row < KEY_ROW_COUNT; row++) {
        // 快速设置行状态
        HAL_GPIO_WritePin(KEY_GPIO_PORT, KEY_ROW_PINS, ~(1 << row));
        __NOP();  // 短暂延时
        
        // 快速读取列状态
        uint16_t col_state = HAL_GPIO_ReadPin(KEY_GPIO_PORT, KEY_COL_PINS);
        matrix_state |= (col_state << (row * KEY_COL_COUNT));
        
        // 快速恢复行状态
        HAL_GPIO_WritePin(KEY_GPIO_PORT, KEY_ROW_PINS, GPIO_PIN_SET);
    }
    
    return matrix_state;
}

// 位操作优化的状态比较
static inline bool IsKeyStateChanged(uint16_t new_state, uint16_t old_state) {
    return (new_state ^ old_state) != 0;
}
```

## 6. 低功耗优化

### 6.1 低功停机模式支持
```c
// 低功耗模式下的按键扫描
typedef enum {
    POWER_MODE_ACTIVE = 0,
    POWER_MODE_SLEEP,
    POWER_MODE_DEEP_SLEEP,
    POWER_MODE_STANDBY
} PowerMode_TypeDef;

// 低功耗扫描配置
typedef struct {
    uint32_t active_scan_interval;    // 活动模式扫描间隔 (ms)
    uint32_t sleep_scan_interval;     // 睡眠模式扫描间隔 (ms)
    uint32_t wake_up_timeout;         // 唤醒超时 (ms)
    PowerMode_TypeDef current_mode;
} LowPowerKeyConfig_TypeDef;

// 低功耗按键扫描
void KeyScan_LowPowerMode(void) {
    static uint32_t last_activity_time = 0;
    uint32_t current_time = HAL_GetTick();
    
    // 检查是否需要进入低功耗模式
    if ((current_time - last_activity_time) > 5000) {  // 5秒无活动
        // 切换到睡眠模式，延长扫描间隔
        KeyTimer_SetInterval(KEY_SCAN_TIMER, 100);  // 100ms扫描间隔
        
        // 配置唤醒源
        HAL_PWR_EnableWakeUpPin(PWR_WAKEUP_PIN1);
        
        // 进入低功耗模式
        HAL_SuspendTick();
        HAL_PWR_EnterSLEEPMode(PWR_MAINREGULATOR_ON, PWR_SLEEPENTRY_WFI);
        HAL_ResumeTick();
        
        // 恢复正常扫描间隔
        KeyTimer_SetInterval(KEY_SCAN_TIMER, 10);
        
        return;
    }
    
    // 正常扫描
    KeyTimer_ScanISR();
    last_activity_time = current_time;
}
```

### 7.1 错误检测

```c
// 按键系统错误类型
typedef enum {
    KEY_ERROR_NONE = 0,
    KEY_ERROR_SCAN_TIMEOUT,
    KEY_ERROR_MATRIX_SHORT,
    KEY_ERROR_HARDWARE_FAULT,
    KEY_ERROR_ISR_OVERFLOW

// 错误处理
typedef struct {
    KeyError_TypeDef last_error;
    uint32_t error_count;
    uint32_t scan_timeout_count;
    uint32_t matrix_short_count;
    uint32_t isr_overflow_count;
} KeyErrorHandler_TypeDef;

// 错误检测函数
bool KeyScan_CheckErrors(void) {
    // 检测矩阵短路
    uint16_t test_state = ReadKeyMatrix();
    if (test_state == 0xFFFF) {  // 所有按键都按下，可能是短路
        g_key_error_handler.last_error = KEY_ERROR_MATRIX_SHORT;
        g_key_error_handler.matrix_short_count++;
        return true;
    }
    
    // 检测扫描超时
    if (HAL_GetTick() - g_key_scan_context.last_scan_time > 100) {
        g_key_error_handler.last_error = KEY_ERROR_SCAN_TIMEOUT;
        g_key_error_handler.scan_timeout_count++;
        return true;
    }
    
    return false;
}
```

## 9. 性能对比

### 9.1 响应时间对比
| 项目 | 轮询方式 | 定时器中断方式 |
|------|----------|----------------|
| 最大响应延迟 | 取决于主循环周期 | 最大10ms |
| 平均响应时间 | 主循环周期的一半 | 5ms |
| 实时性 | 一般 | 良好 |

### 9.2 资源占用对比
| 项目 | 轮询方式 | 定时器中断方式 |
|------|----------|----------------|
| CPU占用率 | 高 | 低 |
| 电源消耗 | 高 | 可进入低功耗模式 |
| 实时性保证 | 无 | 有 |

通过定时器中断驱动的按键系统，我们获得了：
- **更好的实时性**: 最大10ms响应延迟
- **更低的功耗**: 支持低功耗模式
- **更少的CPU占用**: 减少主循环负担
- **更可靠的检测**: 专业化的扫描算法
