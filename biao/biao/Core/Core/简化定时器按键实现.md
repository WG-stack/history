# 简化定时器中断驱动按键扫描实现

## 1. 简化设计需求

根据要求实现：
- **扫描周期**: 50ms
- **防抖**: 无防抖处理
- **长按检测**: 1秒（1000ms）
- **定时器**: 使用现有TIM2
- **兼容性**: 保持与现有接口兼容

## 2. 简化实现方案

### 2.1 简化配置头文件

```c
#ifndef __SIMPLE_KEY_TIMER_H__
#define __SIMPLE_KEY_TIMER_H__

#include "key_driver.h"

// 简化配置参数
#define KEY_SCAN_INTERVAL_MS      50    // 50ms扫描周期
#define KEY_LONG_PRESS_TIME_MS    1000  // 1000ms长按检测
#define KEY_SCAN_COUNTS_PER_LONG  (KEY_LONG_PRESS_TIME_MS / KEY_SCAN_INTERVAL_MS)  // 20次扫描

// 简化状态结构体
typedef struct {
    uint32_t long_press_counter;
    uint16_t last_key_state;
    uint16_t current_key_state;
    uint32_t long_press_start_time;
} SimpleKeyContext_TypeDef;

extern SimpleKeyContext_TypeDef g_key_context;

// 简化接口函数
bool SimpleKeyTimer_Init(void);
void SimpleKeyTimer_Start(void);
void SimpleKeyTimer_Stop(void);
void SimpleKeyTimer_Scan(void);

#endif /* __SIMPLE_KEY_TIMER_H__ */
```

### 2.2 简化实现代码

```c
#include "simple_key_timer.h"
#include "key_driver.h"
#include "tim.h"
#include "stm32l1xx_hal.h"

// 全局上下文
static SimpleKeyContext_TypeDef g_key_context = {0};

// 简化的按键矩阵读取
static uint16_t ReadKeyMatrix_Simple(void) {
    // 使用现有的按键扫描函数
    KeyScanResult_TypeDef result = KeyDriver_Scan();
    
    // 根据按键类型返回对应的矩阵状态
    switch (result.key_type) {
        case KEY_ON_ZERO:
            return 0x0001;
        case KEY_DOWN_ABS_SET:
            return 0x0002;
        case KEY_UP_IN_MM:
            return 0x0004;
        case KEY_TOL:
            return 0x0008;
        case KEY_M:
            return 0x0010;
        case KEY_ABS_ZERO_SWITCH:
            return 0x0020;
        default:
            return 0x0000;
    }
}

// 简化的按键扫描函数
void SimpleKeyTimer_Scan(void) {
    // 1. 读取当前按键状态
    uint16_t current_state = ReadKeyMatrix_Simple();
    
    // 2. 检测按键状态变化
    if (current_state != g_key_context.last_state) {
        // 按键状态发生变化
        
        if (current_state != 0) {
            // 有按键按下
            g_key_context.long_press_counter = 1;
            g_key_context.long_press_start_time = HAL_GetTick();
            
            // 立即发布按键按下事件
            KeyScanResult_TypeDef key_result = {
                .key_type = DetectKeyType(current_state),
                .event = KEY_EVENT_SHORT_PRESS,
                .is_pressed = true
            };
            
            // 调用现有的按键处理函数
            KeyDriver_Process(&key_result);
            
        } else {
            // 按键释放
            if (g_key_context.long_press_counter >= KEY_SCAN_COUNTS_PER_LONG) {
                // 长按事件
                KeyScanResult_TypeDef key_result = {
                    .key_type = DetectKeyType(g_key_context.last_state),
                    .event = KEY_EVENT_LONG_PRESS,
                    .is_pressed = true
                };
                
                KeyDriver_Process(&key_result);
            } else {
                // 短按释放
                KeyScanResult_TypeDef key_result = {
                    .key_type = DetectKeyType(g_key_context.last_state),
                    .event = KEY_EVENT_RELEASE,
                    .is_pressed = false
                };
                
                KeyDriver_Process(&key_result);
            }
            
            g_key_context.long_press_counter = 0;
        }
        
        g_key_context.last_state = current_state;
    } else if (current_state != 0) {
        // 按键持续按下，检测长按
        if (g_key_context.long_press_counter < KEY_SCAN_COUNTS_PER_LONG) {
            g_key_context.long_press_counter++;
            
            if (g_key_context.long_press_counter >= KEY_SCAN_COUNTS_PER_LONG) {
                // 达到长按阈值
                KeyScanResult_TypeDef key_result = {
                    .key_type = DetectKeyType(current_state),
                    .event = KEY_EVENT_LONG_PRESS,
                    .is_pressed = true
                };
                
                KeyDriver_Process(&key_result);
            }
        }
    }
    
    g_key_context.current_key_state = current_state;
}

// 按键类型检测
static Key_TypeDef DetectKeyType(uint16_t matrix_state) {
    switch (matrix_state) {
        case 0x0001: return KEY_ON_ZERO;
        case 0x0002: return KEY_DOWN_ABS_SET;
        case 0x0004: return KEY_UP_IN_MM;
        case 0x0008: return KEY_TOL;
        case 0x0010: return KEY_M;
        case 0x0020: return KEY_ABS_ZERO_SWITCH;
        default: return KEY_NONE;
    }
}

// 定时器中断服务程序
void TIM2_IRQHandler(void) {
    // 清除中断标志
    HAL_TIM_IRQHandler(&htim2);
    
    // 执行按键扫描
    SimpleKeyTimer_Scan();
}

// 简化初始化函数
bool SimpleKeyTimer_Init(void) {
    // 重用现有的TIM2配置
    // 如果TIM2已经被其他功能使用，可以考虑使用TIM3
    
    // 配置TIM2为50ms中断
    __HAL_RCC_TIM2_CLK_ENABLE();
    
    // 配置定时器
    htim2.Instance = TIM2;
    htim2.Init.Prescaler = (SystemCoreClock / 10000) - 1;  // 100us计数
    htim2.Init.CounterMode = TIM_COUNTER_MODE_UP;
    htim2.Init.Period = 50000;  // 50ms = 50000 * 100us
    htim2.Init.ClockDivision = TIM_CLOCKDIVISION_DIV1;
    
    if (HAL_TIM_Base_Init(&htim2) != HAL_OK) {
        return false;
    }
    
    // 启动定时器中断
    if (HAL_TIM_Base_Start_IT(&htim2) != HAL_OK) {
        return false;
    }
    
    // 配置中断优先级
    HAL_NVIC_SetPriority(TIM2_IRQn, 3, 0);
    HAL_NVIC_EnableIRQ(TIM2_IRQn);
    
    return true;
}

void SimpleKeyTimer_Start(void) {
    HAL_TIM_Base_Start_IT(&htim2);
}

void SimpleKeyTimer_Stop(void) {
    HAL_TIM_Base_Stop_IT(&htim2);
}
```

## 6. 注意事项

### 6.1 定时器冲突
如果TIM2已被其他功能使用，可以修改为TIM3：
```c
#define KEY_SCAN_TIMER    TIM3
#define KEY_SCAN_IRQn     TIM3_IRQn
```

### 6.2 中断优先级
确保定时器中断优先级适中，不要影响传感器中断

### 6.3 内存使用
简化设计使用较少的内存，适合资源受限的嵌入式系统
