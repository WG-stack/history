# 中断驱动事件系统优化

## 1. 问题分析

当前系统中传感器数据通过硬件中断被动接收，这对事件驱动架构有特殊要求：

### 1.1 中断特性分析
- **传感器采样**: 通过GPIO中断触发，数据准备好后立即处理
- **按键扫描**: 轮询方式，可以改为中断驱动
- **定时任务**: 通过TIM中断实现周期性操作
- **USB通信**: 通过USB中断处理数据收发

### 1.2 中断安全要求
- 中断服务程序(ISR)必须快速执行
- 不能在ISR中进行复杂的数据处理
- 需要避免ISR中的阻塞操作
- 要考虑中断嵌套和优先级

### 2.3 传感器中断适配

```c
// sensor_sample.c 中断适配
#include "isr_event_handler.h"

// 传感器中断服务程序
void SENSOR_DATA_ISR(void) {
    // 1. 保存中断状态
    uint32_t primask = __get_PRIMASK();
    __disable_irq();
    
    // 2. 读取传感器数据（快速操作）
    volatile Sample_sample_t* sample = &current_sample;
    sample->value = ReadSensorRawData();
    sample->timestamp = HAL_GetTick();
    
    // 3. 发布事件（中断安全）
    Event_TypeDef sensor_event = {
        .type = EVENT_SENSOR_DATA_READY,
        .priority = EVENT_PRIORITY_HIGH,
        .data.sensor_data.raw_value = sample->value,
        .timestamp = sample->timestamp,
        .source_id = SOURCE_ID_SENSOR_ISR
    };
    
    // 使用中断安全的发布函数
    ISR_PostEvent(EVENT_SENSOR_DATA_READY, &sensor_event.data);
    
    // 4. 触发事件处理（非阻塞）
    ISR_EventQueue_TriggerProcessing();
    
    // 5. 恢复中断状态
    if (primask == 0) {
        __enable_irq();
    }
}

// 传感器初始化
void SensorSample_Init_IRQ(void) {
    // 配置传感器中断
    HAL_NVIC_SetPriority(SENSOR_IRQn, 0, 0);  // 最高优先级
    HAL_NVIC_EnableIRQ(SENSOR_IRQn);
    
    // 注册中断处理
    ISR_Context_RegisterHandler(ISR_TYPE_SENSOR_DATA, SENSOR_DATA_ISR);
    
    // 配置传感器GPIO中断
    GPIO_InitTypeDef GPIO_InitStruct = {0};
    GPIO_InitStruct.Pin = SENSOR_DATA_PIN;
    GPIO_InitStruct.Mode = GPIO_MODE_IT_RISING;
    GPIO_InitStruct.Pull = GPIO_NOPULL;
    HAL_GPIO_Init(SENSOR_GPIO_PORT, &GPIO_InitStruct);
}

// 主循环中的事件处理
void SensorSample_MainLoop(void) {
    // 处理传感器数据事件
    Event_TypeDef event;
    while (ISR_EventQueue_Pop(&event)) {
        if (event.type == EVENT_SENSOR_DATA_READY) {
            // 处理传感器数据
            ProcessSensorData(event.data.sensor_data.raw_value);
            
            // 触发后续处理
            Event_TypeDef next_event = {
                .type = EVENT_DATA_PROCESSING,
                .priority = EVENT_PRIORITY_HIGH,
                .timestamp = HAL_GetTick()
            };
            EventManager_PostEvent(&g_event_manager, &next_event);
        }
    }
}
```

### 2.4 按键中断优化

```c
// key_driver.c 中断适配
#include "isr_event_handler.h"

// 按键中断服务程序
void KEY_MATRIX_ISR(void) {
    // 1. 快速读取按键状态
    uint16_t key_state = ReadKeyMatrix();
    
    // 2. 检测按键变化（避免重复触发）
    static uint16_t last_key_state = 0;
    uint16_t key_changes = key_state ^ last_key_state;
    last_key_state = key_state;
    
    if (key_changes) {
        // 3. 发布按键事件
        Event_TypeDef key_event = {
            .type = EVENT_KEY_PRESS,
            .priority = EVENT_PRIORITY_HIGH,
            .data.key_event.key_result.key_type = DetectKeyFromMatrix(key_state),
            .data.key_event.key_result.event = (key_changes & key_state) ? 
                                              KEY_EVENT_PRESS : KEY_EVENT_RELEASE,
            .timestamp = HAL_GetTick(),
            .source_id = SOURCE_ID_KEY_ISR
        };
        
        // 中断安全发布
        ISR_PostEvent(EVENT_KEY_PRESS, &key_event.data);
        
        // 触发事件处理
        ISR_EventQueue_TriggerProcessing();
    }
}

// 按键初始化
void KeyDriver_Init_IRQ(void) {
    // 配置按键矩阵中断
    HAL_NVIC_SetPriority(KEY_IRQn, 1, 0);  // 高优先级
    HAL_NVIC_EnableIRQ(KEY_IRQn);
    
    // 注册中断处理
    ISR_Context_RegisterHandler(ISR_TYPE_KEY_INTERRUPT, KEY_MATRIX_ISR);
    
    // 配置按键GPIO
    GPIO_InitTypeDef GPIO_InitStruct = {0};
    GPIO_InitStruct.Pin = KEY_ROW_PINS | KEY_COL_PINS;
    GPIO_InitStruct.Mode = GPIO_MODE_IT_FALLING;
    GPIO_InitStruct.Pull = GPIO_PULLUP;
    HAL_GPIO_Init(KEY_GPIO_PORT, &GPIO_InitStruct);
```

### 3.2 中断嵌套处理

```c
// 中断嵌套安全的事件处理
typedef struct {
    uint32_t nested_interrupt_count;
    uint32_t critical_section_count;
    uint32_t last_irq_state;
} InterruptSafetyContext_TypeDef;

// 安全的中断嵌套管理
void EnterInterruptSafeSection(void) {
    if (IS_IRQ_ENABLED()) {
        __disable_irq();
        g_interrupt_safety.nested_interrupt_count++;
    }
}

void ExitInterruptSafeSection(void) {
    if (g_interrupt_safety.nested_interrupt_count > 0) {
        g_interrupt_safety.nested_interrupt_count--;
        if (g_interrupt_safety.nested_interrupt_count == 0 && 
            g_interrupt_safety.last_irq_state == 0) {
            __enable_irq();
        }
    }
}

// 中断安全的事件发布
bool ISR_PostEventSafe(EventType_TypeDef type, EventData_TypeDef* data) {
    EnterInterruptSafeSection();
    
    bool result = false;
    if (!g_interrupt_safety.nested_interrupt_count) {
        result = ISR_EventQueue_Push(type, data);
    } else {
        // 嵌套中断时，延迟处理
        result = QueueDeferredEvent(type, data);
    }
    
    ExitInterruptSafeSection();
    return result;
}
```

## 4. 事件处理优化

### 4.1 中断触发的事件处理

```c
// 事件处理优化
typedef struct {
    Event_TypeDef* current_event;
    uint32_t processing_time;
    uint32_t max_processing_time;
    bool is_processing;
} EventProcessor_TypeDef;

// 优化的事件处理函数
bool OptimizedEventHandler(Event_TypeDef* event) {
    uint32_t start_time = HAL_GetTick();
    
    switch (event->type) {
        case EVENT_SENSOR_DATA_READY:
            return HandleSensorDataReady_Optimized(event);
            
        case EVENT_KEY_PRESS:
            return HandleKeyPress_Optimized(event);
            
        case EVENT_DISPLAY_UPDATE:
            return HandleDisplayUpdate_Optimized(event);
            
        default:
            return false;
    }
    
    uint32_t processing_time = HAL_GetTick() - start_time;
    if (processing_time > g_event_processor.max_processing_time) {
        g_event_processor.max_processing_time = processing_time;
    }
    
    return true;
}

// 优化的传感器数据处理
bool HandleSensorDataReady_Optimized(Event_TypeDef* event) {
    // 快速路径：直接处理关键数据
    int32_t raw_value = event->data.sensor_data.raw_value;
    
    // 1. 快速更新数据缓冲区
    g_system_context.current_sample.value = raw_value;
    g_system_context.last_activity_time = HAL_GetTick();
    
    // 2. 触发数据处理（异步）
    Event_TypeDef process_event = {
        .type = EVENT_DATA_PROCESSING,
        .priority = EVENT_PRIORITY_HIGH,
        .timestamp = HAL_GetTick()
    };
    
    // 使用非阻塞方式发布
    return EventManager_PostEvent_NonBlocking(&g_event_manager, &process_event);
}
```

### 4.2 批量事件处理

```c
// 批量事件处理优化
typedef struct {
    Event_TypeDef batch_buffer[EVENT_BATCH_SIZE];
    uint32_t batch_count;
    uint32_t batch_start_time;
} EventBatch_TypeDef;

// 批量处理函数
bool ProcessEventBatch(EventBatch_TypeDef* batch) {
    uint32_t start_time = HAL_GetTick();
    
    for (uint32_t i = 0; i < batch->batch_count; i++) {
        Event_TypeDef* event = &batch->batch_buffer[i];
        
        // 快速处理高优先级事件
        if (event->priority == EVENT_PRIORITY_HIGH) {
            OptimizedEventHandler(event);
        }
    }
    
    // 处理中低优先级事件
    for (uint32_t i = 0; i < batch->batch_count; i++) {
        Event_TypeDef* event = &batch->batch_buffer[i];
        
        if (event->priority != EVENT_PRIORITY_HIGH) {
            OptimizedEventHandler(event);
        }
    }
    
    uint32_t processing_time = HAL_GetTick() - start_time;
    return (processing_time < MAX_BATCH_PROCESSING_TIME);
}

// 主循环中的批量处理
void MainLoop_EventProcessing(void) {
    EventBatch_TypeDef batch;
    
    // 批量获取事件
    uint32_t events_processed = ISR_EventQueue_GetBatch(&batch.batch_buffer, 
                                                      EVENT_BATCH_SIZE);
    
    if (events_processed > 0) {
        batch.batch_count = events_processed;
        batch.batch_start_time = HAL_GetTick();
        
        // 批量处理
        if (!ProcessEventBatch(&batch)) {
            // 处理超时，记录性能问题
            LogPerformanceIssue(PERF_ISSUE_EVENT_PROCESSING_SLOW);
        }
    }
}
```

```c
// 性能监控函数
void PerformanceMonitor_Start(void) {
    HAL_NVIC_EnableIRQ(DWT_IRQn);  // 启用DWT周期计数器
    DWT->CTRL |= DWT_CTRL_CYCCNTENA;
}

uint32_t PerformanceMonitor_GetCycles(void) {
    return DWT->CYCCNT;
}

void PerformanceMonitor_LogISR(uint32_t isr_type, uint32_t start_time, 
                             uint32_t end_time) {
    uint32_t duration = end_time - start_time;
    
    switch (isr_type) {
        case ISR_TYPE_SENSOR_DATA:
            g_performance.sensor_isr_count++;
            g_performance.sensor_isr_total_time += duration;
            if (duration > g_performance.sensor_isr_max_time) {
                g_performance.sensor_isr_max_time = duration;
            }
            break;
            
        case ISR_TYPE_KEY_INTERRUPT:
            g_performance.key_isr_count++;
            g_performance.key_isr_total_time += duration;
            if (duration > g_performance.key_isr_max_time) {
                g_performance.key_isr_max_time = duration;
            }
            break;
    }
}
```

### 6.2 关键注意事项

1. **中断响应时间**: 传感器ISR必须在10μs内完成
2. **内存访问安全**: 所有共享数据必须使用原子操作或临界区保护
3. **事件丢失处理**: 需要处理队列溢出和事件丢失情况
4. **优先级反转**: 确保高优先级事件不会被低优先级中断阻塞
5. **电源管理**: 中断驱动架构对低功耗模式的影响

- 实现微秒级的传感器响应
- 保证系统的实时性和稳定性
- 提供完整的性能监控和调试能力
- 支持系统的低功耗需求
