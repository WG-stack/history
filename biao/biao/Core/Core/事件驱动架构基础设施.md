## 3. 事件队列实现

### 3.1 队列数据结构

```c
#ifndef __EVENT_QUEUE_H__
#define __EVENT_QUEUE_H__

#include "event_data.h"

// 配置参数
#define EVENT_QUEUE_MAX_SIZE  32
#define EVENT_ID_INCREMENT    1

// 事件队列结构体

    bool is_empty;

// 队列操作函数声明
bool EventQueue_Initialize(EventQueue_TypeDef* queue);
```

### 3.2 队列实现代码

```c
#include "event_queue.h"
#include "event_data.h"

// 事件队列实现
bool EventQueue_Initialize(EventQueue_TypeDef* queue) {
    if (queue == NULL) return false;
    
    queue->head = 0;
    queue->tail = 0;
    queue->count = 0;
    queue->next_event_id = 1;
    queue->is_full = false;
    queue->is_empty = true;
    
    return true;
}

bool EventQueue_IsFull(EventQueue_TypeDef* queue) {
    return (queue != NULL) ? queue->is_full : true;
}

bool EventQueue_IsEmpty(EventQueue_TypeDef* queue) {
    return (queue != NULL) ? queue->is_empty : true;
}

bool EventQueue_Push(EventQueue_TypeDef* queue, Event_TypeDef* event) {
    if (queue == NULL || event == NULL || EventQueue_IsFull(queue)) {
        return false;
    }
    
    // 生成事件ID
    event->event_id = queue->next_event_id++;
    
    // 设置时间戳
    event->timestamp = HAL_GetTick();
    
    // 入队
    queue->events[queue->head] = *event;
    queue->head = (queue->head + 1) % EVENT_QUEUE_MAX_SIZE;
    queue->count++;
    
    // 更新状态标志
    queue->is_empty = (queue->count == 0);
    queue->is_full = (queue->count == EVENT_QUEUE_MAX_SIZE);
    
    return true;
}

bool EventQueue_Pop(EventQueue_TypeDef* queue, Event_TypeDef* event) {
    if (queue == NULL || event == NULL || EventQueue_IsEmpty(queue)) {
        return false;
    }
    
    // 出队
    *event = queue->events[queue->tail];
    queue->tail = (queue->tail + 1) % EVENT_QUEUE_MAX_SIZE;
    queue->count--;
    
    // 更新状态标志
    queue->is_full = (queue->count == EVENT_QUEUE_MAX_SIZE);
    queue->is_empty = (queue->count == 0);
    
    return true;
}

uint16_t EventQueue_GetCount(EventQueue_TypeDef* queue) {
    return (queue != NULL) ? queue->count : 0;
}
```

```c
// 事件管理器实现
bool EventManager_Initialize(EventManager_TypeDef* manager) {
    if (manager == NULL) return false;
    
    // 初始化队列
    EventQueue_Initialize(&manager->queue);
    
    // 注册事件处理器
    manager->handler_count = 0;
    manager->processed_events = 0;
    manager->dropped_events = 0;
    
    // 注册默认处理器
    EventManager_RegisterHandler(manager, EVENT_SENSOR_DATA_READY, 
                               HandleSensorDataReady, EVENT_PRIORITY_HIGH);
    EventManager_RegisterHandler(manager, EVENT_KEY_PRESS, 
                               HandleKeyPress, EVENT_PRIORITY_HIGH);
    EventManager_RegisterHandler(manager, EVENT_DISPLAY_UPDATE, 
                               HandleDisplayUpdate, EVENT_PRIORITY_MEDIUM);
    
    return true;
}

bool EventManager_RegisterHandler(EventManager_TypeDef* manager, 
                                EventType_TypeDef type, 
                                EventHandlerFunc handler,
                                uint8_t priority) {
    if (manager == NULL || manager->handler_count >= MAX_EVENT_HANDLERS) {
        return false;
    }
    
    EventHandler_TypeDef* handler_entry = &manager->handlers[manager->handler_count];
    handler_entry->event_type = type;
    handler_entry->handler = handler;
    handler_entry->priority = priority;
    handler_entry->is_enabled = true;
    handler_entry->handler_id = manager->handler_count + 1;
    
    manager->handler_count++;
    
    return true;
}

bool EventManager_PostEvent(EventManager_TypeDef* manager, Event_TypeDef* event) {
    if (manager == NULL || event == NULL) {
        return false;
    }
    
    return EventQueue_Push(&manager->queue, event);
}

bool EventManager_ProcessEvents(EventManager_TypeDef* manager, uint32_t max_events) {
    if (manager == NULL) return false;
    
    uint32_t processed = 0;
    Event_TypeDef event;
    
    while (processed < max_events && !EventQueue_IsEmpty(&manager->queue)) {
        if (EventQueue_Pop(&manager->queue, &event)) {
            // 查找对应的处理器
            for (uint8_t i = 0; i < manager->handler_count; i++) {
                if (manager->handlers[i].is_enabled && 
                    manager->handlers[i].event_type == event.type) {
                    
                    if (manager->handlers[i].handler(&event)) {
                        manager->processed_events++;
                        processed++;
                        break;
                    }
                }
            }
        } else {
            manager->dropped_events++;
        }
    }
    
    return (processed > 0);
}

void EventManager_Run(EventManager_TypeDef* manager) {
    if (manager == NULL) return;
    
    // 处理所有待处理事件
    EventManager_ProcessEvents(manager, EVENT_QUEUE_MAX_SIZE);
}
```

```c
static inline bool ProcessHighPriorityEvents(EventManager_TypeDef* manager) {
    // 优先处理高优先级事件
    // ...

// 批量处理优化
bool ProcessEventBatch(EventManager_TypeDef* manager, uint32_t batch_size) {
    uint32_t processed = 0;
    Event_TypeDef events[batch_size];
    
    // 批量出队
    for (uint32_t i = 0; i < batch_size; i++) {
        if (!EventQueue_Pop(&manager->queue, &events[i])) break;
    }
    
    // 批量处理
    for (uint32_t i = 0; i < processed; i++) {
        ProcessEvent(&events[i]);
    }
    
    return processed;
}
```
